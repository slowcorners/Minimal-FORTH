;	example program printing out the character set to the terminal

        ORG     0x8000      ; set PC to start address of the program

outb:   DB      0
outc:   DB      0

start:  LDI     32          ; write <SPC> into outb
        STA     outb        ; :
next:   LDA     outb        ; :
        OUT                 ; outputs A to terminal (alternatively use STA $8000)
        LDI     8           ; Wait a little
        STA     outc
wait:   DEB     outc
        BNE     wait
        INB     outb        ; increment outb
        BPL     next        ; A < 128? jump back to label 'next'
        JPA     start       ; jump to 'start' address
